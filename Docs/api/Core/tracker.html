Core.tracker

tracker.py — DM-side combat tracker UI for DungeonPy.

Runs a PySimpleGUI initiative table in a daemon thread (in both/dm mode the map occupies the main thread). The tracker displays combatants sorted by initiative, lets the DM manage HP, conditions, and turns, and communicates changes to the MapManager via the GameServer event bus.

Key interactions

  • Token selected on map → handle_server_event highlights the matching row.
  • Turn advanced in tracker → server emits turn_advanced → map applies blue glow.
  • Combatant added/edited → server emits combatant_updated → map redraws token.
  • _squelch_table_event prevents feedback loops when the map triggers a row change.

 44def _render_emoji_png(char: str, size: int = 22) -> bytes | None:
 45    """Render a single emoji as a colour PNG via NotoColorEmoji. Returns None on failure."""
 46    if not _PIL_OK or not os.path.exists(_NOTO_COLOR_EMOJI):
 47        return None
 48    try:
 49        font = PILFont.truetype(_NOTO_COLOR_EMOJI, 109)  # 109 = native bitmap size
 50        dummy = Image.new("RGBA", (1, 1))
 51        bbox = ImageDraw.Draw(dummy).textbbox((0, 0), char, font=font, embedded_color=True)
 52        w = max(1, bbox[2] - bbox[0])
 53        h = max(1, bbox[3] - bbox[1])
 54        img = Image.new("RGBA", (w, h), (0, 0, 0, 0))
 55        ImageDraw.Draw(img).text((-bbox[0], -bbox[1]), char, font=font, embedded_color=True)
 56        img = img.resize((size, size), Image.LANCZOS)
 57        buf = io.BytesIO()
 58        img.save(buf, format="PNG")
 59        return buf.getvalue()
 60    except Exception:
 61        return None

137        """
138        Return {condition_name: png_bytes} for each condition.

145                with open(png_path, 'rb') as f:

168    def _apply_server_event(self, event: dict):
169        """Handle a server event on the GUI thread."""
170        if event.get("type") == "snapshot":
171            # Validate that _selected_index is still in range (e.g. after load_map strips NPCs)
172            if self._selected_index is not None and self._selected_index >= len(self.server.combatants):
173                self._selected_index = None
174                self._clear_form()
175            self.window['-TURN-'].update(str(self.server.turn))
176            self.refresh_table(self._selected_index)
177            state = event.get("state", {})
178            self._sync_map_ui(state.get("map_path"), state.get("map_visible", False))
179            self._rebuild_chat_tabs()
180            return
181
182        action = event.get("action")
183
184        if action == "combatant_updated":
185            self.refresh_table(self._selected_index)
186
187        elif action == "combatant_added":
188            self.refresh_table(self._selected_index)
189
190        elif action == "combatant_removed":
191            self._selected_index = None
192            self.refresh_table()
193
194        elif action == "turn_advanced":
195            self.window['-TURN-'].update(str(event["turn"]))
196            self.refresh_table()
197
198        elif action == "selection_changed" and "selector" not in event:
199            name = event["name"]
200            for i, c in enumerate(self.server.combatants):
201                if c.name == name:
202                    # Only highlight the row if selection actually moved (avoids echo from our own click)
203                    if i != self._selected_index:
204                        self._selected_index = i
205                        self._squelch_table_event = 1
206                        self.window['-TABLE-'].update(select_rows=[i + 1])
207                    self.window['-NAME-'].update(c.name)
208                    self.window['-INITIATIVE-'].update(c.initiative)
209                    self.window['-HP-'].update('' if c.hp is None else c.hp)
210                    self.window['-MAX_HP-'].update('' if c.max_hp is None else c.max_hp)
211                    self.window['-IS_PC-'].update(c.is_pc)
212                    self.window['-SIZE-'].update(c.size)
213                    for cond in self.condition_list:
214                        self.window[f'-COND_{cond}-'].update(cond in c.conditions)
215                    self.window.refresh()
216                    break
217
218        elif action == "selection_cleared" and "selector" not in event:
219            if self._selected_index is not None:
220                self._selected_index = None
221                self._squelch_table_event = 1
222                self.window['-TABLE-'].update(select_rows=[])
223            self._clear_form()
224
225        elif action == "player_connected":
226            color = event.get("color", "white")
227            self._connected_players[event["name"]] = {
228                "select": False, "move": False, "color": color,
229            }
230            self._known_players[event["name"]] = color
231            self._refresh_players_table()
232            self.refresh_table(self._selected_index)
233
234        elif action == "player_disconnected":
235            self._connected_players.pop(event["name"], None)
236            self._refresh_players_table()
237            self.refresh_table(self._selected_index)
238
239        elif action == "player_lock_changed":
240            name = event["name"]
241            if name in self._connected_players:
242                lock_type = event.get("lock_type", "move")
243                self._connected_players[name][lock_type] = event["locked"]
244                self._refresh_players_table()
245
246        elif action == "map_loaded":
247            self._sync_map_ui(event.get("path"), True)
248
249        elif action == "map_visibility_changed":
250            self._sync_map_ui(self._map_path, event.get("visible", False))
251
252        elif action in ("combatant_added", "combatant_removed", "combatant_updated"):
253            self._rebuild_chat_tabs()
254
255        elif action == "chat_message":
256            sender  = event.get("from", "DM")
257            to_name = event.get("to")
258            text    = event.get("text", "")
259            # Determine which PC tab to post to
260            pc_name = to_name if to_name else sender
261            if self._chat:
262                self._chat.receive(pc_name, sender, text)
263            if sender != "DM":
264                try:
265                    self.window['-CHAT_NOTIFY-'].update('● new message')
266                except Exception:
267                    pass

312    def _start_notes_edit(self, event):
313        """Open an inline Entry widget over the Notes cell on double-click."""
314        tree = self.window['-TABLE-'].Widget
315        if tree.identify_region(event.x, event.y) != 'cell':
316            return
317        if tree.identify_column(event.x) != '#4':  # Notes is the 4th data column
318            return
319        item = tree.identify_row(event.y)
320        if not item:
321            return
322        children = list(tree.get_children())
323        idx = children.index(item)
324        if idx == 0:  # blank deselection row
325            return
326        combatant = self.server.combatants[idx - 1]
327
328        bbox = tree.bbox(item, '#4')
329        if not bbox:
330            return
331        x, y, w, h = bbox
332
333        var = tk.StringVar(value=combatant.notes or '')
334        entry = tk.Entry(tree, textvariable=var)
335        entry.place(x=x, y=y, width=w, height=h)
336        entry.select_range(0, 'end')
337        entry.focus()
338
339        def commit(e=None):
340            new_val = var.get()
341            entry.destroy()
342            self._submit({"action": "update_combatant", "name": combatant.name,
343                          "fields": {"notes": new_val}})
344
345        entry.bind('<Return>', commit)
346        entry.bind('<FocusOut>', commit)
347        entry.bind('<Escape>', lambda e: entry.destroy())

353    def _make_condition_strip(self, conditions: list):
354        """Return an ImageTk.PhotoImage strip of condition icons, or None."""
355        if not _PIL_OK or not conditions:
356            return None
357        size, gap = CONDITION_ICON_SIZE, 2
358        imgs = []
359        for cond in conditions:
360            raw = self.condition_images.get(cond)
361            if raw:
362                imgs.append(Image.open(io.BytesIO(raw)).convert("RGBA").resize((size, size), Image.LANCZOS))
363        if not imgs:
364            return None
365        total_w = len(imgs) * size + (len(imgs) - 1) * gap
366        strip = Image.new("RGB", (total_w, size), (255, 255, 255))
367        for i, img in enumerate(imgs):
368            strip.paste(img, (i * (size + gap), 0), img)
369        return ImageTk.PhotoImage(strip)

458    def refresh_table(self, selected_index=None):
459        data = [['', '', '', '']]  # blank row for deselection
460        row_conditions = [[]]  # parallel list of condition lists per row
461        row_names      = ['']   # raw combatant names (no "→ " prefix) for color lookup
462        for i, c in enumerate(self.server.combatants):
463            name = f"→ {c.name}" if i == self.server.active_index else c.name
464            timer_parts = [f"{_COND_ABBREV.get(cond, cond[:3])}:{exp[0]}@{exp[1]}"
465                           for cond, exp in sorted(c.condition_timers.items())]
466            notes_display = c.notes + (" [" + ", ".join(timer_parts) + "]" if timer_parts else "")
467            data.append([name, c.initiative, '' if c.hp is None else c.hp, notes_display])
468            row_conditions.append(list(c.conditions))
469            row_names.append(c.name)
470
471        if selected_index is not None and 0 <= selected_index < len(self.server.combatants):
472            self._squelch_table_event = 2  # update clears then reselects — two TABLE events expected
473            self.window['-TABLE-'].update(values=data, select_rows=[selected_index + 1])
474        else:
475            self._squelch_table_event = 1  # deselect fires one TABLE event
476            self.window['-TABLE-'].update(values=data, select_rows=[])
477
478        tree = self.window['-TABLE-'].Widget
479        tree.tag_configure('dead', font=('Helvetica', 16, 'overstrike'))
480        for info in self._connected_players.values():
481            c = info.get("color", "white")
482            try:
483                r, g, b = (v >> 8 for v in tree.winfo_rgb(c))
484                dimmed = f'#{(r + 255) // 2:02x}{(g + 255) // 2:02x}{(b + 255) // 2:02x}'
485            except Exception:
486                dimmed = c
487            tree.tag_configure(f'pc_{c}', background=dimmed, foreground='black')
488        new_photos = {}
489        for item_id, conditions, cname in zip(tree.get_children(), row_conditions, row_names):
490            connected = self._connected_players.get(cname)
491            color_tag = (f'pc_{connected["color"]}',) if connected else ()
492            tags = (('dead',) + color_tag) if 'Dead' in conditions else color_tag
493            photo = self._make_condition_strip(conditions)
494            tree.item(item_id, text='', image=photo if photo else '', tags=tags)
495            if photo:
496                new_photos[item_id] = photo
497        self._table_photos = new_photos  # replace; old refs released

503    def handle_event(self, event, values, dir_path):
504        if event == sg.WIN_CLOSED:
505            return
506
507        if event == '-TABLE-' and self._squelch_table_event > 0:
508            self._squelch_table_event -= 1
509            return
510
511        if self.verbose:
512            log(f"[Tracker] Event: {event}")
513
514        if event == '-TABLE-':
515            try:
516                if values['-TABLE-']:
517                    row_index = values['-TABLE-'][0]
518                    if self.verbose:
519                        log(f"[Tracker] Handling row selection: {row_index}")
520                    if row_index == 0:
521                        self._selected_index = None
522                        self._submit({"action": "clear_selection"})
523                        self.window['-TABLE-'].update(select_rows=[])
524                        self._clear_form()
525                    else:
526                        self._selected_index = row_index - 1
527                        c = self.server.combatants[self._selected_index]
528                        if self.verbose:
529                            log(f"[Tracker] Selected index = {self._selected_index}, Combatant: {c}")

776    def run_gui(self, dir_path):
777        """Build and run the tracker GUI event loop.  Blocks until the window is closed.
778
779        Args:
780            dir_path: Base directory used to resolve save/load file paths.
781        """
782        layout = self.build_gui_layout()
783        self.window = sg.Window('D&D Initiative Tracker', layout, resizable=True, finalize=True)
784
785        # Intercept window close at Tk level — read_all_windows destroys the window
786        # before we see WIN_CLOSED, so we replace WM_DELETE_WINDOW entirely.
787        # We must NOT call popup from here (nested mainloop → unstable).
788        # Just post an event; the read_all_windows loop shows the popup safely.
789        self.window.TKroot.protocol(
790            "WM_DELETE_WINDOW",
791            lambda: self.window.write_event_value('-CLOSE_REQUESTED-', None),
792        )
793
794        tree = self.window['-TABLE-'].Widget
795        tree.bind('<Double-Button-1>', self._start_notes_edit)
796        if _PIL_OK:
797            tree.configure(show='tree headings')
798            tree.column('#0', width=150, stretch=False, anchor='w')
799            tree.heading('#0', text='')
800
801        self.refresh_table()
802        self._squelch_table_event = 0  # initial populate doesn't fire a TABLE event; reset to avoid eating first click
803        # Restore map UI state from server (save may have been loaded before window opened)
804        self._sync_map_ui(self.server.map_path, self.server.map_visible)
805        self.window['-SIGHT_RADIUS-'].update(value=self.server.visibility_radius)
806
807        # Open chat window at startup
808        self._chat = ChatWindow(submit_fn=self._submit)
809        self._chat.open(self._pc_names())
810
811        while True:
812            win, event, values = sg.read_all_windows(timeout=100)
813
814            if win is None:
815                continue  # timeout with no event
816
817            if win == self.window:
818                if event == '-CLOSE_REQUESTED-':
819                    if sg.popup_yes_no('Are you sure you want to quit DungeonPy?',
820                                       title='Quit', keep_on_top=True) == 'Yes':
821                        break
822                    continue
823                if event == sg.WIN_CLOSED:
824                    break  # fallback if window destroyed externally
825                if event == 'SERVER_EVENT':
826                    self._apply_server_event(values[event])
827                    continue
828                self.handle_event(event, values, dir_path)
829
830            elif win == (self._chat.window if self._chat else None):
831                keep = self._chat.handle_event(event, values)
832                if not keep:
833                    self.window['Toggle Chat'].update(text='Open Chat')
834                if not self._chat._unread:
835                    self.window['-CHAT_NOTIFY-'].update('')
836
837        if self._chat:
838            self._chat.close()
839        self.window.close()
840        os._exit(0)

169    def _apply_server_event(self, event: dict):
170        """Handle a server event on the GUI thread."""
171        if event.get("type") == "snapshot":
172            # Validate that _selected_index is still in range (e.g. after load_map strips NPCs)
173            if self._selected_index is not None and self._selected_index >= len(self.server.combatants):
174                self._selected_index = None
175                self._clear_form()
176            self.window['-TURN-'].update(str(self.server.turn))
177            self.refresh_table(self._selected_index)
178            state = event.get("state", {})
179            self._sync_map_ui(state.get("map_path"), state.get("map_visible", False))
180            self._rebuild_chat_tabs()
181            return
182
183        action = event.get("action")
184
185        if action == "combatant_updated":
186            self.refresh_table(self._selected_index)
187
188        elif action == "combatant_added":
189            self.refresh_table(self._selected_index)
190
191        elif action == "combatant_removed":
192            self._selected_index = None
193            self.refresh_table()
194
195        elif action == "turn_advanced":
196            self.window['-TURN-'].update(str(event["turn"]))
197            self.refresh_table()
198
199        elif action == "selection_changed" and "selector" not in event:
200            name = event["name"]
201            for i, c in enumerate(self.server.combatants):
202                if c.name == name:
203                    # Only highlight the row if selection actually moved (avoids echo from our own click)
204                    if i != self._selected_index:
205                        self._selected_index = i
206                        self._squelch_table_event = 1
207                        self.window['-TABLE-'].update(select_rows=[i + 1])
208                    self.window['-NAME-'].update(c.name)
209                    self.window['-INITIATIVE-'].update(c.initiative)
210                    self.window['-HP-'].update('' if c.hp is None else c.hp)
211                    self.window['-MAX_HP-'].update('' if c.max_hp is None else c.max_hp)
212                    self.window['-IS_PC-'].update(c.is_pc)
213                    self.window['-SIZE-'].update(c.size)
214                    for cond in self.condition_list:
215                        self.window[f'-COND_{cond}-'].update(cond in c.conditions)
216                    self.window.refresh()
217                    break
218
219        elif action == "selection_cleared" and "selector" not in event:
220            if self._selected_index is not None:
221                self._selected_index = None
222                self._squelch_table_event = 1
223                self.window['-TABLE-'].update(select_rows=[])
224            self._clear_form()
225
226        elif action == "player_connected":
227            color = event.get("color", "white")
228            self._connected_players[event["name"]] = {
229                "select": False, "move": False, "color": color,
230            }
231            self._known_players[event["name"]] = color
232            self._refresh_players_table()
233            self.refresh_table(self._selected_index)
234
235        elif action == "player_disconnected":
236            self._connected_players.pop(event["name"], None)
237            self._refresh_players_table()
238            self.refresh_table(self._selected_index)
239
240        elif action == "player_lock_changed":
241            name = event["name"]
242            if name in self._connected_players:
243                lock_type = event.get("lock_type", "move")
244                self._connected_players[name][lock_type] = event["locked"]
245                self._refresh_players_table()
246
247        elif action == "map_loaded":
248            self._sync_map_ui(event.get("path"), True)
249
250        elif action == "map_visibility_changed":
251            self._sync_map_ui(self._map_path, event.get("visible", False))
252
253        elif action in ("combatant_added", "combatant_removed", "combatant_updated"):
254            self._rebuild_chat_tabs()
255
256        elif action == "chat_message":
257            sender  = event.get("from", "DM")
258            to_name = event.get("to")
259            text    = event.get("text", "")
260            # Determine which PC tab to post to
261            pc_name = to_name if to_name else sender
262            if self._chat:

264            if sender != "DM":

267                except Exception:

777    def run_gui(self, dir_path):
778        """Build and run the tracker GUI event loop.  Blocks until the window is closed.

786        # Intercept window close at Tk level — read_all_windows destroys the window
787        # before we see WIN_CLOSED, so we replace WM_DELETE_WINDOW entirely.
788        # We must NOT call popup from here (nested mainloop → unstable).
789        # Just post an event; the read_all_windows loop shows the popup safely.
790        self.window.TKroot.protocol(
791            "WM_DELETE_WINDOW",
792            lambda: self.window.write_event_value('-CLOSE_REQUESTED-', None),
793        )
794
795        tree = self.window['-TABLE-'].Widget
796        tree.bind('<Double-Button-1>', self._start_notes_edit)
797        if _PIL_OK:
798            tree.configure(show='tree headings')
799            tree.column('#0', width=150, stretch=False, anchor='w')
800            tree.heading('#0', text='')
801
802        self.refresh_table()
803        self._squelch_table_event = 0  # initial populate doesn't fire a TABLE event; reset to avoid eating first click
804        # Restore map UI state from server (save may have been loaded before window opened)
805        self._sync_map_ui(self.server.map_path, self.server.map_visible)
806        self.window['-SIGHT_RADIUS-'].update(value=self.server.visibility_radius)
807
808        # Open chat window at startup
809        self._chat = ChatWindow(submit_fn=self._submit)
810        self._chat.open(self._pc_names())
811
812        while True:
813            win, event, values = sg.read_all_windows(timeout=100)
814
815            if win is None:
816                continue  # timeout with no event
817
818            if win == self.window:
819                if event == '-CLOSE_REQUESTED-':
820                    if sg.popup_yes_no('Are you sure you want to quit DungeonPy?',
821                                       title='Quit', keep_on_top=True) == 'Yes':
822                        break
823                    continue
824                if event == sg.WIN_CLOSED:
825                    break  # fallback if window destroyed externally
826                if event == 'SERVER_EVENT':
827                    self._apply_server_event(values[event])
828                    continue
829                self.handle_event(event, values, dir_path)
830
831            elif win == (self._chat.window if self._chat else None):
832                keep = self._chat.handle_event(event, values)
833                if not keep:
834                    self.window['Toggle Chat'].update(text='Open Chat')
835                if not self._chat._unread:
836                    self.window['-CHAT_NOTIFY-'].update('')

838        if self._chat:
839            self._chat.close()
840        self.window.close()
841        os._exit(0)
